// Free ROI sample
// =========================================================================
// 1. LOAD ADMINISTRATIVE BOUNDARIES FOR MALAWI
// =========================================================================

// Load Level 0 for the entire country boundary
var malawiCountry = ee.FeatureCollection("FAO/GAUL/2015/level0")
  .filter(ee.Filter.eq('ADM0_NAME', 'Malawi'));

// Load Level 2 for district-level boundaries
var malawiDistricts = ee.FeatureCollection("FAO/GAUL/2015/level2")
  .filter(ee.Filter.eq('ADM0_NAME', 'Malawi'));

// Print district attributes to the console (useful for checking names like Karonga, Blantyre, etc.)
print("Malawi Districts Properties:", malawiDistricts.limit(5));


// =========================================================================
// 2. DESIGN STYLING & VISUALIZATION
// =========================================================================

// Style for the country outline (Thick red border, transparent fill)
var countryStyle = {
  color: 'FF0000',      // Red outline
  width: 3,             // Thick border
  fillColor: '00000000' // Transparent fill
};

// Style for the districts (Thin blue border, light semi-transparent blue fill)
var districtStyle = {
  color: '0055FF',      // Blue outline
  width: 1,             // Thin border
  fillColor: '0055FF1A' // 10% opacity blue fill
};


// =========================================================================
// 3. DISPLAY ON THE MAP
// =========================================================================

// Reset map view layers
Map.clear();

// Center the map view dynamically on Malawi (zoom level 7 is perfect for the country scale)
Map.centerObject(malawiCountry, 7);

// Add layers to the map workspace
Map.addLayer(malawiDistricts.style(districtStyle), {}, "Malawi Districts (Level 2)");
Map.addLayer(malawiCountry.style(countryStyle), {}, "Malawi Country Boundary (Level 0)");

Free Supervised Training Sample

// =========================================================================
// 1. LOAD THE ENTIRE CONTINENT OF AFRICA AS THE STUDY AREA
// =========================================================================
// Load the global large scale international boundaries dataset
var continents = ee.FeatureCollection("USDOS/LSIB/2017");

// Filter out the African continent geometry
var africa = continents.filter(ee.Filter.eq('wld_rgn', 'Africa'));

// Center the map view on Africa (Zoom level 3 or 4 covers the continent)
Map.clear();
Map.centerObject(africa, 3);
Map.addLayer(africa, {color: 'grey'}, 'Africa Boundary', false);


// =========================================================================
// 2. LANDSAT 9 IMAGE COLLECTION & SCALING
// =========================================================================
// Landsat 9 Collection 2, Tier 1, Level 2 Top-of-Atmosphere / Surface Reflectance
var image = ee.ImageCollection("LANDSAT/LC09/C02/T1_L2")
            .filterBounds(africa)
            .filterDate('2024-01-01', '2024-12-31')
            // Using a slightly higher cloud tolerance since parts of Central Africa are highly cloudy
            .filterMetadata('CLOUD_COVER', 'less_than', 30) 
            .median() // Creates the continental cloud-free composite
            .clip(africa);

// Apply the correct scale factor and offset formula for Landsat 2 Surface Reflectance bands
var opticalBands = image.select('SR_B.').multiply(0.0000275).add(-0.2);
var thermalBands = image.select('ST_B.*').multiply(0.00341802).add(149.0);

// Re-combine the scaled data back into a single image
var scaledImage = image.addBands(opticalBands, null, true)
                       .addBands(thermalBands, null, true);

// Select the necessary spectral bands for analysis
var bands = ['SR_B1', 'SR_B2', 'SR_B3', 'SR_B4', 'SR_B5', 'SR_B6', 'SR_B7'];
var input = scaledImage.select(bands);

// 1. LOAD THE ENTIRE CONTINENT OF AFRICA AS THE STUDY AREA
// =========================================================================
// Load the global large scale international boundaries dataset
var continents = ee.FeatureCollection("USDOS/LSIB/2017");

// Filter out the African continent geometry
var africa = continents.filter(ee.Filter.eq('wld_rgn', 'Africa'));

// Center the map view on Africa (Zoom level 3 or 4 covers the continent)
Map.clear();
Map.centerObject(africa, 3);
Map.addLayer(africa, {color: 'grey'}, 'Africa Boundary', false);


// =========================================================================
// 2. LANDSAT 9 IMAGE COLLECTION & SCALING
// =========================================================================
// Landsat 9 Collection 2, Tier 1, Level 2 Top-of-Atmosphere / Surface Reflectance
var image = ee.ImageCollection("LANDSAT/LC09/C02/T1_L2")
            .filterBounds(africa)
            .filterDate('2024-01-01', '2024-12-31')
            // Using a slightly higher cloud tolerance since parts of Central Africa are highly cloudy
            .filterMetadata('CLOUD_COVER', 'less_than', 30) 
            .median() // Creates the continental cloud-free composite
            .clip(africa);

// Apply the correct scale factor and offset formula for Landsat 2 Surface Reflectance bands
var opticalBands = image.select('SR_B.').multiply(0.0000275).add(-0.2);
var thermalBands = image.select('ST_B.*').multiply(0.00341802).add(149.0);

// Re-combine the scaled data back into a single image
var scaledImage = image.addBands(opticalBands, null, true)
                       .addBands(thermalBands, null, true);

// Select the necessary spectral bands for analysis
var bands = ['SR_B1', 'SR_B2', 'SR_B3', 'SR_B4', 'SR_B5', 'SR_B6', 'SR_B7'];
var input = scaledImage.select(bands);


// =========================================================================
// 3. TRAINING DATA & SAMPLING CONFIGURATION
// =========================================================================
// CRITICAL: Ensure your 'water', 'vegetation', 'bareland', and 'builtup' 
// feature collections contain points distributed across various countries in Africa.
var samples = water.merge(vegetation).merge(bareland).merge(builtup);

print("Total training points submitted:", samples.size());

// Sampling pixel values underneath the scattered point clusters
var trainImage = input.sampleRegions({
  collection: samples,
  properties: ['class'],
  scale: 30,
  tileScale: 4 // Essential! Prevents out-of-memory errors on massive continental operations
});


// =========================================================================
// 4. DATA SPLIT & CLASSIFICATION (CART MODEL)
// =========================================================================
var trainData = trainImage.randomColumn();

var trainSet = trainData.filter(ee.Filter.lessThan('random', 0.8));
var testSet = trainData.filter(ee.Filter.greaterThanOrEquals('random', 0.8));

var classifier = ee.Classifier.smileCart().train({
  features: trainSet,
  classProperty: 'class',
  inputProperties: bands
});

// Run the classification across the whole continent footprint
var classified = input.classify(classifier);

// Display the results
Map.addLayer(classified, {min: 0, max: 3, palette: ['blue', 'green', 'brown', 'red']}, 'Africa Classified LULC');


// =========================================================================
// 5. EXPORT THE CONTINENTAL MAP TO GOOGLE DRIVE
// =========================================================================
Export.image.toDrive({
  image: classified,
  description: 'Africa_LULC_2024_Classification',
  folder: 'GEE_Africa_Classification', 
  region: africa.geometry(), // Converts feature collection boundary to geometry format
  scale: 100,                // NOTE: Changed from 30m to 100m to prevent export crashes. See below.
  maxPixels: 1e13             
});

Free Unsupervised LULC Sample

// =========================================================================
// 1. DEFINE STUDY AREA (MALAWI)
// =========================================================================
// Load FAO Global Administrative Unit Layers (Country Level)
var countries = ee.FeatureCollection("FAO/GAUL/2015/level0");

// Filter explicitly for Malawi
var malawi = countries.filter(ee.Filter.eq('ADM0_NAME', 'Malawi'));

Map.clear();
Map.centerObject(malawi, 7); // Zoom level 7 is perfect for Malawi
Map.addLayer(malawi, {color: 'grey'}, 'Malawi Boundary', false);

// =========================================================================
// 2. LANDSAT 9 COMPOSITE FOR MALAWI (2024)
// =========================================================================
var collection = ee.ImageCollection("LANDSAT/LC09/C02/T1_L2")
            .filterBounds(malawi)
            .filterDate('2024-01-01', '2024-12-31');

// Print collection size to verify imagery is found
print("Number of Landsat 9 scenes found over Malawi:", collection.size());

// Create the median cloud-free composite and clip to Malawi
var image = collection.median().clip(malawi);

// Apply correct Landsat 9 scale factors to the optical surface reflectance bands
var opticalBands = image.select('SR_B.*').multiply(0.0000275).add(-0.2);

// Specify the exact spectral bands needed for our clustering model
var bands = ['SR_B1', 'SR_B2', 'SR_B3', 'SR_B4', 'SR_B5', 'SR_B6', 'SR_B7'];
var input = opticalBands.select(bands);

// =========================================================================
// 3. GENERATE RANDOM TRAINING PIXELS (HIGH RESOLUTION FOR MALAWI)
// =========================================================================
var trainingPixels = input.sample({
  region: malawi,
  scale: 30,           // 30m native Landsat resolution works perfectly for country scale
  numPixels: 5000,     // 5,000 points provides an excellent statistical distribution for Malawi
  geometries: true,
  tileScale: 4         
});

// =========================================================================
// 4. TRAIN THE UNSUPERVISED CLUSTERER (K-MEANS)
// =========================================================================
var numberOfClusters = 4; // Target classes: Water, Vegetation, Bareland, Built-up
var clusterer = ee.Clusterer.wekaKMeans(numberOfClusters).train(trainingPixels);

// =========================================================================
// 5. CLASSIFY MALAWI & VISUALIZE
// =========================================================================
var clustered = input.cluster(clusterer);

// Palette layout mapping assignments
// Note: Check your map to see which color corresponds to your land cover,
// and rearrange the colors array below if needed.
var paletteColors = ['blue', 'green', 'orange', 'red']; 

Map.addLayer(clustered, {min: 0, max: 3, palette: paletteColors}, 'Malawi Unsupervised Clusters');

// =========================================================================
// 6. EXPORT CLUSTERED IMAGE TO GOOGLE DRIVE (NATIVE 30M)
// =========================================================================
Export.image.toDrive({
  image: clustered,
  description: 'Malawi_Unsupervised_LULC_2024',
  folder: 'GEE_Malawi_Unsupervised',
  region: malawi.geometry(),
  scale: 30,           // Export at full 30-meter resolution
  maxPixels: 1e13
});
