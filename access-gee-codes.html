Unsupervised Clustering Base

// Free unsupervised clustering starter
// =========================================================================
var image = ee.Image('COPERNICUS/S2_SR/20230705T084609_20230705T085230_T36NUR');
var bands = ['B4', 'B3', 'B2', 'B8'];
var clusterer = ee.Clusterer.wekaKMeans(5).train(image.select(bands));
var clusters = image.select(bands).cluster(clusterer);
Map.addLayer(clusters.randomVisualizer(), {}, 'Unsupervised Clusters');
